function bodyBadgeClass(bodyId) { if (bodyId === 'council') return 'mb-council'; if (bodyId === 'recreation') return 'mb-recreation'; if (bodyId === 'museum') return 'mb-museum'; if (bodyId === 'cemetery') return 'mb-cemetery'; return 'mb-council'; } function renderCalendar() { const today = new Date(); today.setHours(0, 0, 0, 0); const upcoming = [...ALL_MEETINGS] .filter(m => !m.cancelled && m.date) .filter(m => { const d = new Date(`${m.date}T00:00:00`); return !Number.isNaN(d.getTime()) && d >= today; }) .sort((a, b) => String(a.date || '').localeCompare(String(b.date || ''))); const byMonth = {}; upcoming.forEach(m => { const d = new Date(`${m.date}T00:00:00`); const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`; const label = d.toLocaleDateString('en-CA', { year: 'numeric', month: 'long' }); if (!byMonth[key]) byMonth[key] = { label, items: [] }; byMonth[key].items.push(m); }); const months = Object.keys(byMonth).sort(); document.getElementById('calendar-wrap').innerHTML = months.map(key => { const month = byMonth[key]; return `
${escHtml(month.label)}
${month.items.map(m => `
${escHtml(fmtDate(m.date))}
${escHtml(m.body || 'Meeting')}
${m.agenda_url ? `Agenda` : ''} ${m.minutes_url ? `Minutes` : ''} ${m.package_url ? `Package` : ''}
`).join('')}
`; }).join(''); }
